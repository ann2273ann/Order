<script>
let order = [];
let total = 0;

function addItem(name, price){
    order.push(name);
    total += price;
    render();
}

function render(){
    document.getElementById("order").innerHTML = order.join("<br>");
    document.getElementById("total").innerText = "總金額：" + total;
}

function submitOrder(){
    alert("訂單已送出！\n" + order.join(", "));
    order = [];
    total = 0;
    render();
}
